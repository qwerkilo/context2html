### 27. D3.js 数据可视化

> **🎯 效果**：D3.js 创建自定义数据可视化——力导向图、弦图、旭日图、桑基图等 ECharts 不原生支持的图表类型。交互式悬浮提示 + 动画过渡。

#### 前置依赖

D3.js 库需要独立下载到本地 `libs/` 目录。两种加载方式：

**离线加载（推荐）**：

```bash
# PowerShell（Windows）
Invoke-WebRequest -Uri "https://d3js.org/d3.v7.min.js" -OutFile "libs/d3.min.js"

# macOS / Linux
mkdir -p libs && curl -Lo libs/d3.min.js https://d3js.org/d3.v7.min.js
```

```html
<script src="libs/d3.min.js"></script>
```

**CDN 加载（备选）**：

```html
<script src="https://d3js.org/d3.v7.min.js"></script>
```

#### 27a — 力导向图（网络关系）

适合展示实体间的关系网络（组织结构、知识图谱、社交网络）。

```html
<figure class="svg-fig">
  <div id="chart-force" style="width:100%;height:450px;margin:1rem 0;"></div>
</figure>
```

```css
#chart-force { background: var(--surface); border-radius: var(--radius); position: relative; overflow: hidden; }
.force-node { stroke: #fff; stroke-width: 1.5px; cursor: pointer; transition: r 0.2s ease; }
.force-node:hover { stroke-width: 3px; }
.force-label { font-size: 11px; fill: var(--text); pointer-events: none; }
.force-link { stroke: var(--border); stroke-width: 1px; stroke-opacity: 0.5; }
```

```js
(function(){
  var el = document.getElementById('chart-force');
  if (!el || typeof d3 === 'undefined') return;
  var w = el.clientWidth, h = 450;
  var svg = d3.select(el).append('svg').attr('width',w).attr('height',h);

  // ===== 数据（可修改） =====
  var nodes = [
    { id: 'A', label: '节点 A', group: 1, r: 20 },
    { id: 'B', label: '节点 B', group: 1, r: 16 },
    { id: 'C', label: '节点 C', group: 2, r: 14 },
    { id: 'D', label: '节点 D', group: 2, r: 18 },
  ];
  var links = [
    { source: 'A', target: 'B' },
    { source: 'A', target: 'C' },
    { source: 'B', target: 'C' },
    { source: 'C', target: 'D' },
  ];

  var colors = d3.scaleOrdinal(d3.schemeTableau10);
  var sim = d3.forceSimulation(nodes)
    .force('link', d3.forceLink(links).id(function(d){return d.id;}).distance(100))
    .force('charge', d3.forceManyBody().strength(-200))
    .force('center', d3.forceCenter(w/2, h/2));

  var link = svg.selectAll('.force-link').data(links).join('line').attr('class','force-link');
  var node = svg.selectAll('.force-node').data(nodes).join('circle')
    .attr('class','force-node').attr('r',function(d){return d.r;})
    .attr('fill',function(d){return colors(d.group);})
    .call(d3.drag().on('start',function(e,d){if(!e.active)sim.alphaTarget(0.3).restart();d.fx=d.x;d.fy=d.y;})
      .on('drag',function(e,d){d.fx=e.x;d.fy=e.y;})
      .on('end',function(e,d){if(!e.active)sim.alphaTarget(0);d.fx=null;d.fy=null;}));
  var label = svg.selectAll('.force-label').data(nodes).join('text')
    .attr('class','force-label').text(function(d){return d.label;})
    .attr('text-anchor','middle').attr('dy','0.35em');

  sim.on('tick',function(){
    link.attr('x1',function(d){return d.source.x;}).attr('y1',function(d){return d.source.y;})
      .attr('x2',function(d){return d.target.x;}).attr('y2',function(d){return d.target.y;});
    node.attr('cx',function(d){return d.x;}).attr('cy',function(d){return d.y;});
    label.attr('x',function(d){return d.x;}).attr('y',function(d){return d.y;});
  });
})();
```

#### 27b — 旭日图（层次占比）

适合展示多层级的占比构成（预算分配、组织结构、分类占比）。

```html
<figure class="svg-fig">
  <div id="chart-sunburst" style="width:100%;height:400px;margin:1rem 0;"></div>
</figure>
```

```css
#chart-sunburst { background: var(--surface); border-radius: var(--radius); }
.sunburst-arc { stroke: #fff; stroke-width: 1.5px; cursor: pointer; transition: opacity 0.2s ease; }
.sunburst-arc:hover { opacity: 0.8; }
```

```js
// D3.js 旭日图需要 d3-hierarchy
// CDN: <script src="https://d3js.org/d3.v7.min.js"></script> 已包含 hierarchy
(function(){
  var el = document.getElementById('chart-sunburst');
  if (!el || typeof d3 === 'undefined') return;
  var w = el.clientWidth, h = 400, cx = w/2, cy = h/2, r = Math.min(w,h)/2 - 10;

  // ===== 层次数据（可修改） =====
  var data = {
    name: 'root',
    children: [
      { name: '类别 A', value: 40,
        children: [{name:'A1',value:25},{name:'A2',value:15}] },
      { name: '类别 B', value: 35,
        children: [{name:'B1',value:20},{name:'B2',value:15}] },
      { name: '类别 C', value: 25,
        children: [{name:'C1',value:15},{name:'C2',value:10}] },
    ]
  };

  var root = d3.hierarchy(data).sum(function(d){return d.value;});
  var arc = d3.arc().startAngle(function(d){return d.x0;}).endAngle(function(d){return d.x1;})
    .innerRadius(function(d){return d.y0;}).outerRadius(function(d){return d.y1;});
  var color = d3.scaleOrdinal(d3.schemeTableau10);
  var svg = d3.select(el).append('svg').attr('width',w).attr('height',h)
    .append('g').attr('transform','translate('+cx+','+cy+')');

  var partition = d3.partition().size([2 * Math.PI, r]);
  partition(root);

  svg.selectAll('path').data(root.descendants()).join('path')
    .attr('class','sunburst-arc').attr('d',arc)
    .attr('fill',function(d){ while(d.depth>1)d=d.parent; return color(d.data.name); })
    .append('title').text(function(d){ return d.data.name + '\n' + d.value + '%'; });
})();
```

#### 27c — 桑基图（流量/流向）

适合展示能量流、资金流、用户转化路径等流量数据。

```html
<figure class="svg-fig">
  <div id="chart-sankey" style="width:100%;height:350px;margin:1rem 0;"></div>
</figure>
```

```css
#chart-sankey { background: var(--surface); border-radius: var(--radius); }
```

```js
// D3.js 桑基图需要 d3-sankey 扩展
// CDN: <script src="https://cdn.jsdelivr.net/npm/d3-sankey@0.12.3/dist/d3-sankey.min.js"></script>
(function(){
  var el = document.getElementById('chart-sankey');
  if (!el || typeof d3 === 'undefined' || typeof d3.sankey === 'undefined') return;
  var w = el.clientWidth, h = 350;

  // ===== 流量数据（可修改） =====
  var data = {
    nodes: [{name:'起点'},{name:'路径A'},{name:'路径B'},{name:'终点C'},{name:'终点D'}],
    links: [{source:0,target:1,value:60},{source:0,target:2,value:40},
            {source:1,target:3,value:35},{source:1,target:4,value:25},
            {source:2,target:3,value:15},{source:2,target:4,value:25}]
  };

  var sankey = d3.sankey().nodeWidth(20).nodePadding(12).extent([[10,10],[w-10,h-10]]);
  var {nodes,links} = sankey({nodes:data.nodes.map(d=>Object.assign({},d)),links:data.links.map(d=>Object.assign({},d))});
  var color = d3.scaleOrdinal(d3.schemeTableau10);
  var svg = d3.select(el).append('svg').attr('width',w).attr('height',h);

  svg.selectAll('.sankey-link').data(links).join('path')
    .attr('d',d3.sankeyLinkHorizontal()).attr('fill','none')
    .attr('stroke',function(d){return color(d.source.index);}).attr('stroke-width',function(d){return Math.max(1,d.width);}).attr('stroke-opacity',0.4);
  svg.selectAll('.sankey-node').data(nodes).join('rect')
    .attr('x',function(d){return d.x0;}).attr('y',function(d){return d.y0;})
    .attr('width',function(d){return d.x1-d.x0;}).attr('height',function(d){return d.y1-d.y0;})
    .attr('fill',function(d){return color(d.index);}).attr('stroke','#fff');
  svg.selectAll('.sankey-label').data(nodes).join('text')
    .attr('x',function(d){return d.x0-6;}).attr('y',function(d){return (d.y0+d.y1)/2;})
    .attr('dy','0.35em').attr('text-anchor','end').attr('font-size','12px')
    .text(function(d){return d.name;});
})();
```

#### 降级说明

| D3 库不存在 | 组件静默不渲染（在 `if (!el || typeof d3 === 'undefined') return;` 处退出） |
| d3-sankey 未加载（桑基图） | 桑基图静默退出，其他子组件不受影响 |
| 响应式 | 使用 `clientWidth` + `clientHeight` 获取容器尺寸，窗口 resize 需触发 `location.reload()` |
| 主题适配 | D3 使用 SVG 原生样式 + CSS 变量，颜色通过 `d3.scaleOrdinal(d3.schemeTableau10)` 自动分配 |
